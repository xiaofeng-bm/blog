给定一个可包含重复数字的序列，返回所有不重复的全排列。

示例:
```md
输入: [1,1,2]
输出:
[
  [1,1,2],
  [1,2,1],
  [2,1,1]
]
```


## 实现代码
```js
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permuteUnique = function (nums) {
  nums.sort((a, b) => {
    return a - b;
  });
  const len = nums.length;
  const result = [];
  const cur = [];
  const visited = [];
  dfs(0);
  function dfs(n) {
    if (n === len) {
      result.push([...cur]);
      return;
    }
    for (let i = 0; i < len; i++) {
      if ((i > 0 && nums[i] === nums[i - 1] && !visited[i - 1]) || visited[i]) {
        continue;
      }
      visited[i] = true;
      cur.push(nums[i]);
      dfs(n + 1);
      cur.pop();
      visited[i] = false;
    }
  }
  return result;
};

// 空间复杂度：On
```
