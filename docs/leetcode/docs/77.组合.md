给定两个整数 n 和 k，返回 1 ... n 中所有可能的 k 个数的组合。

示例:
```md
输入: n = 4, k = 2
输出:
[
  [2,4],
  [3,4],
  [2,3],
  [1,2],
  [1,3],
  [1,4],
]
```
## 代码
回溯套路都差不多，这题就直接写答案吧，加一些注释
```js
/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
var combine = function (n, k) {
  // 结果数组
  const result = [];
  // 当前结果数组
  const cur = [];
  dfs(1);
  function dfs(index) {
    // 当前数组长度===k时，就需要推入结果数组
    if (cur.length === k) {
      result.push([...cur]);
      return;
    }
    for (let i = index; i <= n; i++) {
      cur.push(i);
      dfs(i + 1);
      cur.pop();
    }
  }
  return result;
};
```
